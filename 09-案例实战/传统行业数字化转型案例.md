# 案例实战 - 传统行业数字化转型案例

## 一、案例概述

### 1.1 案例背景

某大型制造业企业，拥有 10 万+ 员工，年销售额 100 亿+，传统 IT 系统老旧，无法支撑数字化转型需求。

### 1.2 业务痛点

| 痛点 | 说明 |
|------|------|
| 系统老旧 | 90 年代的 ERP 系统，无法扩展 |
| 数据孤岛 | 各系统独立，数据不互通 |
| 响应缓慢 | 系统响应时间 > 10s |
| 维护困难 | 系统复杂，维护成本高 |
| 创新受限 | 无法快速响应市场变化 |

### 1.3 转型目标

1. 系统现代化：从传统架构迁移到云架构
2. 数据集成：打通各系统数据孤岛
3. 性能提升：响应时间 < 2s
4. 成本降低：IT 成本降低 30%
5. 创新加速：缩短产品上市周期

```python
        self.app.route('/api/user/info', methods=['GET'])(self.get_user_info)
        self.app.route('/api/user/update', methods=['PUT'])(self.update_user)
    
    def register(self):
        """用户注册"""
        data = request.get_json()
        
        # 验证数据
        if not self.validate_register_data(data):
            return jsonify({'error': 'Invalid data'}), 400
        
        # 检查用户是否已存在
        if self.user_exists(data['username']):
            return jsonify({'error': 'User already exists'}), 400

        hashed_password = bcrypt.hashpw(data['password'].encode(), bcrypt.gensalt())
        
        # 创建用户

            'email': data['email'],
            'phone': data['phone']
        })

            'user_id': user['id'],

        if not bcrypt.checkpw(data['password'].encode(), user['password'].encode()):
            return jsonify({'error': 'Invalid password'}), 401
        
        # 生成 token
        token = self.generate_token(user['id'])
        
        return jsonify({
            'user_id': user['id'],
            'token': token
        })

        """生成 token"""
        payload = {
```

```python
# ERP 数据迁移
import pandas as pd
from sqlalchemy import create_engine

class ERPMigrator:
    def __init__(self, old_db, new_db):
        self.old_db_engine = create_engine(old_db)
        self.new_db_engine = create_engine(new_db)
    
    def migrate_customer_data(self):
        """迁移客户数据"""
        # 从旧系统读取
        old_customers = pd.read_sql('SELECT * FROM customers', self.old_db_engine)
        
        # 数据转换
        new_customers = pd.DataFrame({
            'customer_id': old_customers['cust_id'],
            'customer_name': old_customers['cust_name'],
            'customer_type': self.map_customer_type(old_customers['cust_type']),
            'created_at': pd.to_datetime(old_customers['create_date']),
            'updated_at': pd.to_datetime('now')
        })
        
        # 写入新系统
        new_customers.to_sql('customers', self.new_db_engine, if_exists='append', index=False)
        
        return len(new_customers)
    
    def migrate_order_data(self):
        """迁移订单数据"""
        # 从旧系统读取
        old_orders = pd.read_sql('SELECT * FROM orders', self.old_db_engine)
        
        # 数据转换
        new_orders = pd.DataFrame({
            'order_id': old_orders['order_no'],
            'customer_id': old_orders['cust_id'],
            'order_amount': old_orders['total_amount'],
            'order_status': self.map_order_status(old_orders['status']),
            'created_at': pd.to_datetime(old_orders['create_date']),
            'updated_at': pd.to_datetime('now')
        })
        
        # 写入新系统
        new_orders.to_sql('orders', self.new_db_engine, if_exists='append', index=False)
        
        return len(new_orders)
    
    def migrate_product_data(self):
        """迁移产品数据"""
        # 从旧系统读取
        old_products = pd.read_sql('SELECT * FROM products', self.old_db_engine)
        
        # 数据转换
        new_products = pd.DataFrame({
            'product_id': old_products['prod_no'],
            'product_name': old_products['prod_name'],
            'product_category': old_products['category'],
            'product_price': old_products['unit_price'],
            'created_at': pd.to_datetime(old_orders['create_date']),
            'updated_at': pd.to_datetime('now')
        })
        
        # 写入新系统
        new_products.to_sql('products', self.new_db_engine, if_exists='append', index=False)
        
        return len(new_products)
    
    def validate_migration(self, table_name, old_count, new_count):
        """验证迁移数据"""
        if old_count == new_count:
            print(f"{table_name}: 迁移成功，记录数: {new_count}")
            return True
        else:
            print(f"{table_name}: 迁移失败，旧系统记录数: {old_count}, 新系统记录数: {new_count}")
            return False
```

```python
# 设备数据采集
from pymodbus.client.sync import ModbusTcpClient
import time

class EquipmentDataCollector:
    def __init__(self, plc_ip, port=502):
        self.client = ModbusTcpClient(plc_ip, port=port)
    
    def connect(self):
        """连接 PLC"""
        return self.client.connect()
    
    def disconnect(self):
        """断开连接"""
        self.client.close()
    
    def collect_equipment_data(self, register_start, register_count):
        """采集设备数据"""
        # 读取寄存器
        result = self.client.read_holding_registers(register_start, register_count, unit=1)
        
        if not result.registers:
            return None
        
        # 解析数据
        data = {
            'temperature': result.registers[0] / 10,
            'pressure': result.registers[1] / 10,
            'speed': result.registers[2],
            'status': result.registers[3],
            'timestamp': int(time.time())
        }
        
        return data
    
    def continuous_collect(self, interval=1):
        """持续采集数据"""
        while True:
            data = self.collect_equipment_data(0, 10)
            
            if data:
                # 发送到云端
                self.send_to_cloud(data)
            
            time.sleep(interval)

        response = requests.post(url, json=data)
```

### 5.1 预测性维护

**场景描述：**
通过采集设备数据，使用 AI 模型预测设备故障，提前进行维护，减少停机时间。

**技术方案：**
```python
# 预测性维护
import pandas as pd
from sklearn.ensemble import RandomForestClassifier
from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score

class PredictiveMaintenance:
    def __init__(self):
        self.model = None
    
    def train_model(self, data_path):
        """训练模型"""
        # 加载数据
        data = pd.read_csv(data_path)
        
        # 特征工程
        features = ['temperature', 'vibration', 'pressure', 'runtime']
        X = data[features]
        y = data['failure']
        
        # 划分训练集和测试集
        X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2)
        
        # 训练模型
        self.model = RandomForestClassifier(n_estimators=100, random_state=42)
        self.model.fit(X_train, y_train)
        
        # 评估模型
        y_pred = self.model.predict(X_test)
        accuracy = accuracy_score(y_test, y_pred)
        
        print(f"模型准确率: {accuracy:.2%}")
        
        return accuracy
    
    def predict_failure(self, sensor_data):
        """预测设备故障"""
        if not self.model:
            raise Exception("模型未训练")
        
        # 预测
        probability = self.model.predict_proba([sensor_data])[0][1]
        
        # 判断
        if probability > 0.8:
            return "高危，需要立即维护"
        elif probability > 0.6:
            return "中危，建议尽快维护"
        elif probability > 0.4:
            return "低危，建议定期检查"
        else:
            return "正常，无需维护"
    
    def get_feature_importance(self):
        """获取特征重要性"""
        if not self.model:
            raise Exception("模型未训练")
        
        feature_names = ['temperature', 'vibration', 'pressure', 'runtime']
        importance = self.model.feature_importances_
        
        return dict(zip(feature_names, importance))

# 使用示例
pm = PredictiveMaintenance()
pm.train_model('equipment_data.csv')

# 预测设备故障
sensor_data = {
    'temperature': 75,
    'vibration': 12,
    'pressure': 150,
    'runtime': 1000
}

result = pm.predict_failure(sensor_data)
print(result)
```

### 5.2 智能质检

**场景描述：**
使用计算机视觉技术，自动检测产品质量，提高质检效率。

**技术方案：**
```python
# 智能质检
import cv2
import numpy as np
from tensorflow.keras.models import load_model

class QualityInspection:
    def __init__(self, model_path):
        self.model = load_model(model_path)
        self.class_names = ['合格', '不合格']
    
    def inspect_image(self, image_path):
        """检测图片质量"""
        # 加载图片
        image = cv2.imread(image_path)
        
        # 预处理
        processed_image = self.preprocess_image(image)
        
        # 预测
        prediction = self.model.predict(processed_image)
        class_index = np.argmax(prediction)
        confidence = prediction[0][class_index]
        
        return {
            'class': self.class_names[class_index],
            'confidence': float(confidence)
        }
    
    def preprocess_image(self, image):
        """预处理图片"""
        # 调整大小
        image = cv2.resize(image, (224, 224))
        
        # 归一化
        image = image / 255.0
        
        # 扩展维度
        image = np.expand_dims(image, axis=0)
        
        return image
    
    def batch_inspect(self, image_paths):
        """批量检测"""
        results = []
        
        for image_path in image_paths:
            result = self.inspect_image(image_path)
            results.append({
                'image_path': image_path,
                'result': result
            })
        
        return results

# 使用示例
qi = QualityInspection('quality_model.h5')

# 单个检测

print(result)

# 批量检测

results = qi.batch_inspect(image_paths)
print(results)
```
