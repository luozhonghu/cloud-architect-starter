## 一、电商行业特点

### 1.1 业务特点

#### 高并发
- 大促活动（双11、618）流量爆发

- 24 小时运营，不能停机

#### 快速变化

- 用户行为多变

#### 数据驱动
- 用户画像精准营销

```python
# 使用 Redis + Lua 实现库存扣减
# 避免超卖
def deduct_stock(product_id, quantity):
    script = """
    local stock = tonumber(redis.call('GET', KEYS[1]))
    if stock >= tonumber(ARGV[1]) then
        redis.call('DECRBY', KEYS[1], ARGV[1])
        return 1
    else
        return 0
    end
    """
    result = redis.eval(script, 1, f'stock:{product_id}', quantity)
    if result == 0:
        raise Exception('库存不足')
    return True
```

class PaymentService:
    def create_payment(self, order_no, amount, payment_method):
        """创建支付"""
        # 1. 创建支付订单
        payment = self.create_payment_order(order_no, amount, payment_method)
        # 2. 调用第三方支付
        result = self.call_third_party(payment)
        # 3. 返回支付信息
        return result

    def payment_callback(self, payment_no, status):
        """支付回调"""
        # 1. 验证签名
        if not self.verify_sign(payment_no, status):
            return False
        # 2. 更新支付状态
        self.update_payment_status(payment_no, status)
        # 3. 触发后续业务
        if status == 'SUCCESS':
            self.trigger_order_paid(payment_no)
        return True

```python
import jwt
from datetime import datetime, timedelta

def generate_token(user_id):
    """生成 JWT Token"""
    payload = {
        'user_id': user_id,
        'exp': datetime.utcnow() + timedelta(days=7)
    }
    token = jwt.encode(payload, 'secret_key', algorithm='HS256')
    return token

def verify_token(token):
    """验证 Token"""
    try:
        payload = jwt.decode(token, 'secret_key', algorithms=['HS256'])
        return payload['user_id']
    except jwt.ExpiredSignatureError:
        return None
```

def search_products(keyword, page=1, size=10):
    """搜索商品"""
    query = {
        "query": {
            "match": {
                "name": keyword
            }
        },
        "from": (page - 1) * size,
        "size": size,
        "sort": [

#### 限流降级
```python
# 令牌桶限流
class RateLimiter:
    def __init__(self, rate, capacity):
        self.rate = rate
        self.capacity = capacity
        self.tokens = capacity
        self.last_time = time.time()

    def allow(self):
        now = time.time()
        elapsed = now - self.last_time
        self.tokens += elapsed * self.rate
        self.tokens = min(self.tokens, self.capacity)
        self.last_time = now

        if self.tokens >= 1:
            self.tokens -= 1
            return True
        return False
```
