# 视频直播行业云解决方案

## 1. 行业概述

### 1.1 业务特点

视频直播行业具有以下核心特点：

| 特点 | 说明 | 技术挑战 |
|------|------|----------|
| 高并发 | 同时在线用户数可能达到百万级 | 需要强大的并发处理能力 |
| 低延迟 | 实时互动要求延迟低于500ms | 需要全球CDN加速和优化 |
| 大带宽 | 视频传输消耗大量带宽 | 需要成本优化策略 |
| 高可用 | 业务中断会导致严重损失 | 需要多地域容灾架构 |
| 弹性伸缩 | 流量波动巨大（活动期间10倍+） | 需要自动伸缩能力 |

### 1.2 业务模式

- **娱乐直播**：秀场直播、游戏直播
- **电商直播**：带货直播、产品展示
- **教育直播**：在线课程、企业培训
- **会议直播**：线上发布会、远程会议

```go
package main

import (
    "github.com/gorilla/websocket"
    "github.com/go-redis/redis/v8"
    "context"
)

type DanmuService struct {
    upgrader    websocket.Upgrader
    redisClient *redis.Client
    kafka       *kafka.Producer
}

type DanmuMessage struct {
    RoomID    string `json:"room_id"`
    UserID    string `json:"user_id"`
    Content   string `json:"content"`
    Timestamp int64  `json:"timestamp"`
    Color     string `json:"color"`
}

func (s *DanmuService) HandleWebSocket(w http.ResponseWriter, r *http.Request) {
    conn, err := s.upgrader.Upgrade(w, r, nil)
    if err != nil {
        log.Println("WebSocket upgrade error:", err)
        return
    }
    defer conn.Close()
    
    // 房间ID
    roomID := r.URL.Query().Get("room_id")
    
    // 将连接加入房间
    room := GetRoom(roomID)
    room.Join(conn)
    
    // 处理消息
    for {
        var msg DanmuMessage
        if err := conn.ReadJSON(&msg); err != nil {
            break
        }
        
        // 敏感词过滤
        if s.isSensitive(msg.Content) {
            continue
        }
        
        // 限流控制
        if !s.checkRateLimit(msg.UserID) {
            continue
        }
        
        // 发送到Kafka
        s.kafka.Send("danmu", msg)
        
        // 广播到房间
        room.Broadcast(msg)
    }
    
    room.Leave(conn)
}

// 敏感词过滤
func (s *DanmuService) isSensitive(content string) bool {
    sensitiveWords := []string{"违禁词1", "违禁词2"}
    for _, word := range sensitiveWords {
        if strings.Contains(content, word) {
            return true
        }
    }
    return false
}

// 限流控制（Redis）
func (s *DanmuService) checkRateLimit(userID string) bool {
    ctx := context.Background()
    key := fmt.Sprintf("danmu:rate:%s", userID)
    
    count, _ := s.redisClient.Incr(ctx, key).Result()
    if count == 1 {
        s.redisClient.Expire(ctx, key, time.Minute)
    }
    
    // 每分钟最多10条
    return count <= 10
}
```

### 3.5 礼物打赏系统

#### 系统流程

```

    │
    ▼

    │
    ▼
发送礼物特效消息（WebSocket）
    │
    ▼

    │
    ▼

    │
    ▼
触发排行榜更新
```

#### 礼物服务实现

```java
@Service
public class GiftService {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    @Autowired
    private UserBalanceMapper userBalanceMapper;
    
    @Autowired
    private GiftRecordMapper giftRecordMapper;
    
    @Autowired
    private WebSocketHandler webSocketHandler;
    
    /**
     * 发送礼物
     */
    @Transactional
    public GiftResult sendGift(SendGiftRequest request) {
        String userId = request.getUserId();
        String anchorId = request.getAnchorId();
        String roomId = request.getRoomId();
        String giftId = request.getGiftId();
        int count = request.getCount();
        
        // 1. 获取礼物信息
        Gift gift = getGiftInfo(giftId);
        
        // 2. 扣除用户余额（原子操作）
        boolean success = deductBalance(userId, gift.getPrice() * count);
        if (!success) {
            return GiftResult.fail("余额不足");
        }
        
        // 3. 记录礼物流水
        GiftRecord record = new GiftRecord();
        record.setUserId(userId);
        record.setAnchorId(anchorId);
        record.setRoomId(roomId);
        record.setGiftId(giftId);
        record.setCount(count);
        record.setAmount(gift.getPrice() * count);
        giftRecordMapper.insert(record);
        
        // 4. 更新主播收益
        updateAnchorIncome(anchorId, gift.getPrice() * count);
        
        // 5. 更新房间礼物榜
        updateRoomGiftRank(roomId, userId, gift.getPrice() * count);
        
        // 6. 推送礼物特效消息
        GiftMessage message = new GiftMessage();
        message.setUserId(userId);
        message.setGiftId(giftId);
        message.setCount(count);
        message.setAnimation(gift.getAnimation());
        webSocketHandler.broadcastToRoom(roomId, message);
        
        // 7. 触发连击特效
        if (count >= 5) {
            triggerComboEffect(roomId, userId, giftId, count);
        }
        
        return GiftResult.success();
    }
    
    /**
     * 扣除余额（使用Lua保证原子性）
     */
    private boolean deductBalance(String userId, long amount) {
        String luaScript = 
            "local key = KEYS[1] " +
            "local amount = tonumber(ARGV[1]) " +
            "local current = redis.call('GET', key) " +
            "if not current then " +
            "    return 0 " +
            "end " +
            "current = tonumber(current) " +
            "if current < amount then " +
            "    return 0 " +
            "end " +
            "redis.call('DECRBY', key, amount) " +
            "return 1";
        
        DefaultRedisScript<Long> redisScript = new DefaultRedisScript<>();
        redisScript.setScriptText(luaScript);
        redisScript.setResultType(Long.class);
        
        Long result = redisTemplate.execute(
            redisScript,
            Collections.singletonList("user:balance:" + userId),
            String.valueOf(amount)
        );
        
        return result != null && result == 1;
    }
    
    /**
     * 更新房间礼物榜（Redis Sorted Set）
     */
    private void updateRoomGiftRank(String roomId, String userId, long amount) {
        String key = "room:gift:rank:" + roomId;
        redisTemplate.opsForZSet().incrementScore(key, userId, amount);
    }
    
    /**
     * 获取房间礼物榜Top 10
     */
    public List<RankItem> getRoomGiftRank(String roomId) {
        String key = "room:gift:rank:" + roomId;
        Set<ZSetOperations.TypedTuple<Object>> result = 
            redisTemplate.opsForZSet().reverseRangeWithScores(key, 0, 9);
        
        return result.stream().map(tuple -> {
            RankItem item = new RankItem();
            item.setUserId(tuple.getValue().toString());
            item.setScore(tuple.getScore().longValue());
            return item;
        }).collect(Collectors.toList());
    }
}
```

```javascript
// 根据网络状况自动调整码率
class AdaptiveBitrate {
    constructor() {
        this.currentBitrate = 1200000; // 初始1.2Mbps
        this.lastSpeed = 0;
    }
    
    adjustBitrate(networkSpeed) {
        // 网络速度（bps）
        const speed = networkSpeed * 8;
        
        // 使用当前码率的80%作为安全阈值
        const targetBitrate = Math.floor(speed * 0.8);
        
        // 码率范围：300kbps - 5000kbps
        const minBitrate = 300000;
        const maxBitrate = 5000000;
        
        this.currentBitrate = Math.max(minBitrate, 
                                       Math.min(maxBitrate, targetBitrate));
        
        return this.currentBitrate;
    }
}
```

```python
            ]

        self.oss_client.put_bucket_lifecycle('live-bucket', rule)
    
    def archive_old_videos(self):
        """归档旧视频到归档存储"""
        # 7天前的视频归档
        cutoff_date = datetime.now() - timedelta(days=7)
        prefix = 'live/'
        
        for obj in oss2.ObjectIterator(self.oss_client, prefix=prefix):
            if obj.last_modified < cutoff_date:
                self.oss_client.update_object_meta(
                    'live-bucket',
                    obj.key,
                    {'x-oss-storage-class': oss2.BUCKET_STORAGE_CLASS_ARCHIVE}
```

### 5.2 鉴权方案

#### 直播推流鉴权

```java
public class LiveAuthService {
    
    /**
     * 生成推流URL（带鉴权）
     * 阿里云鉴权方式：A鉴权（URL鉴权）
     */
    public String generateAuthUrl(String domain, String appName, 
                                  String streamName, String key) {
        long timestamp = System.currentTimeMillis() / 1000 + 3600; // 1小时有效
        String uuid = UUID.randomUUID().toString().replace("-", "");
        
        // 生成鉴权字符串
        String authString = String.format("%s-%s-%d-%s", 
            appName, streamName, timestamp, uuid);
        
        // MD5加密
        String sign = md5(authString + key);
        
        // 组装URL
        return String.format("rtmp://%s/%s/%s?auth_key=%d-%s-%s",
            domain, appName, streamName, timestamp, uuid, sign);
    }
    
    private String md5(String input) {
        try {
            MessageDigest md = MessageDigest.getInstance("MD5");
            byte[] digest = md.digest(input.getBytes());
            return DatatypeConverter.printHexBinary(digest).toLowerCase();
        } catch (Exception e) {
            throw new RuntimeException(e);
        }
    }
}
```
