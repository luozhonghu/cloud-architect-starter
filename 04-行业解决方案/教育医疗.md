# 行业解决方案 - 教育医疗

## 一、教育医疗云化概述

### 1.1 教育行业云化特点

教育行业云化具有以下特点：
- **季节性强**：开学季、考试季流量激增
- **高并发访问**：选课、查分、报名等高并发场景
- **数据敏感性**：学生信息、成绩数据敏感
- **多地域分布**：学校分布在全国各地
- **成本敏感**：教育机构预算有限

### 1.2 医疗行业云化特点

医疗行业云化具有以下特点：
- **高安全性要求**：患者隐私数据极其敏感
- **高可用性要求**：医疗系统不能中断
- **数据量大**：医学影像、电子病历数据量大
- **实时性要求**：医疗诊断需要实时处理
- **合规要求严格**：需要满足等保、HIPAA等要求

### 1.3 教育医疗上云挑战

| 挑战 | 说明 | 解决方案 |
|------|------|---------|
| 数据安全 | 学生/患者数据敏感 | 数据加密、访问控制 |
| 高并发 | 选课/挂号高并发 | 自动伸缩、缓存 |
| 数据量大 | 医学影像/电子病历 | 分布式存储、归档 |
| 合规要求 | 等保/ HIPAA | 等保合规服务 |
| 成本控制 | 预算有限 | 成本优化、资源复用 |

# 选课系统核心逻辑
import redis
import time

class CourseSelection:
    def __init__(self):
        self.redis = redis.Redis(host='localhost', port=6379)
    
    def select_course(self, student_id, course_id):
        """学生选课"""
        # 检查课程容量
        capacity_key = f"course:{course_id}:capacity"
        enrolled_key = f"course:{course_id}:enrolled"
        
        capacity = self.redis.get(capacity_key)
        enrolled = self.redis.scard(enrolled_key)
        
        if int(enrolled) >= int(capacity):
            return False, "课程已满"
        
        # 检查是否已选
        student_courses_key = f"student:{student_id}:courses"
        if self.redis.sismember(student_courses_key, course_id):
            return False, "已选该课程"
        
        # 原子操作：添加选课记录
        pipe = self.redis.pipeline()
        pipe.sadd(enrolled_key, student_id)
        pipe.sadd(student_courses_key, course_id)
        pipe.incr(f"course:{course_id}:select_count")
        result = pipe.execute()
        
        if result[0]:
            # 发送通知
            self.send_notification(student_id, course_id)
            return True, "选课成功"
        else:
            return False, "选课失败"
    
    def send_notification(self, student_id, course_id):

#### 2.2.3 在线考试

**场景描述：**

**技术架构：**
```python
# 考试系统核心逻辑
class OnlineExam:
    def __init__(self):
        self.redis = redis.Redis(host='localhost', port=6379)
    
    def start_exam(self, student_id, exam_id):
        """开始考试"""
        # 生成考试令牌
        token = f"exam:{exam_id}:{student_id}:{int(time.time())}"
        
        # 设置考试时间
        exam_key = f"exam:{exam_id}"
        exam_info = self.redis.hgetall(exam_key)
        duration = int(exam_info[b'duration'])
        
        # 设置过期时间
        self.redis.setex(token, duration, "active")
        
        return token
    
    def submit_answer(self, student_id, exam_id, question_id, answer):
        """提交答案"""
        answer_key = f"exam:{exam_id}:{student_id}:answers"
        self.redis.hset(answer_key, question_id, answer)
        
        # 记录答题时间
        timestamp = int(time.time())
        self.redis.hset(f"exam:{exam_id}:{student_id}:times", question_id, timestamp)
        
        return True
    
    def auto_grade(self, student_id, exam_id):
        """自动阅卷"""
        exam_key = f"exam:{exam_id}"
        questions = self.redis.hgetall(f"{exam_key}:questions")
        
        answer_key = f"exam:{exam_id}:{student_id}:answers"
        student_answers = self.redis.hgetall(answer_key)
        
        score = 0
        for question_id, correct_answer in questions.items():
            question_id = question_id.decode()
            correct_answer = correct_answer.decode()
            
            student_answer = student_answers.get(question_id.encode())
            if student_answer and student_answer.decode() == correct_answer:
                score += int(self.redis.hget(f"{exam_key}:scores", question_id))
        
        # 保存成绩
        self.redis.set(f"exam:{exam_id}:{student_id}:score", score)
        
        return score
```

# 电子病历系统核心逻辑
class EMRSystem:
    def __init__(self):
        self.db = MySQLConnection()
        self.kms = KMSClient()
    
    def create_patient(self, patient_info):
        """创建患者档案"""
        # 数据脱敏
        masked_id_card = self.mask_id_card(patient_info['id_card'])
        masked_phone = self.mask_phone(patient_info['phone'])
        
        # 数据加密
        encrypted_info = self.encrypt_data({
            'name': patient_info['name'],
            'id_card': patient_info['id_card'],
            'phone': patient_info['phone']
        })
        
        # 保存到数据库
        sql = """
        INSERT INTO patients 
        (name, id_card, phone, encrypted_info, created_at) 
        VALUES (%s, %s, %s, %s, NOW())
        """
        self.db.execute(sql, (
            patient_info['name'],
            masked_id_card,
            masked_phone,
            encrypted_info
        ))
        
        return self.db.lastrowid
    
    def mask_id_card(self, id_card):
        """身份证号脱敏"""
        return id_card[:6] + '********' + id_card[-4:]
    
    def mask_phone(self, phone):
        """手机号脱敏"""
        return phone[:3] + '****' + phone[-4:]
    
    def encrypt_data(self, data):
        """数据加密"""
        json_data = json.dumps(data)
        return self.kms.encrypt(json_data)
    
    def decrypt_data(self, encrypted_data):
        """数据解密"""
        json_data = self.kms.decrypt(encrypted_data)
        return json.loads(json_data)
```

# 医学影像存储系统
class MedicalImageStorage:
    def __init__(self):
        self.oss = OSSClient()
        self.db = MySQLConnection()
    
    def upload_image(self, patient_id, image_type, image_path):
        """上传医学影像"""
        # 生成唯一文件名
        filename = f"{patient_id}_{image_type}_{int(time.time())}.dcm"
        
        # 上传到OSS
        object_key = f"medical_images/{patient_id}/{filename}"
        self.oss.upload_file(object_key, image_path)
        
        # 记录到数据库
        sql = """
        INSERT INTO medical_images 
        (patient_id, image_type, object_key, upload_time) 
        VALUES (%s, %s, %s, NOW())
        """
        self.db.execute(sql, (patient_id, image_type, object_key))
        
        return object_key
    
    def get_image(self, image_id):
        """获取医学影像"""
        sql = "SELECT * FROM medical_images WHERE id = %s"
        image_info = self.db.fetchone(sql, (image_id,))
        
        if image_info:
            # 生成临时访问URL
            url = self.oss.sign_url(image_info['object_key'], 3600)
            return url
        else:
            return None
    
    def share_image(self, image_id, doctor_id, duration=24):
        """分享医学影像给医生"""
        sql = "SELECT * FROM medical_images WHERE id = %s"
        image_info = self.db.fetchone(sql, (image_id,))
        
        if image_info:
            # 生成分享URL
            url = self.oss.sign_url(image_info['object_key'], duration * 3600)
            
            # 记录分享
            share_sql = """
            INSERT INTO image_shares 
            (image_id, doctor_id, share_time, expire_time) 
            VALUES (%s, %s, NOW(), DATE_ADD(NOW(), INTERVAL %s HOUR))
            """
            self.db.execute(share_sql, (image_id, doctor_id, duration))
            
            return url
        else:
            return None

**技术架构：**

# 挂号系统核心逻辑
class OnlineRegistration:
    def __init__(self):
        self.redis = redis.Redis(host='localhost', port=6379)
        self.db = MySQLConnection()
    
    def get_available_slots(self, doctor_id, date):
        """获取可用号源"""
        # 查询医生排班
        schedule_key = f"doctor:{doctor_id}:schedule:{date}"
        schedule = self.redis.hgetall(schedule_key)
        
        if not schedule:
            # 从数据库加载
            sql = """
            SELECT * FROM doctor_schedules 
            WHERE doctor_id = %s AND schedule_date = %s
            """
            schedules = self.db.fetchall(sql, (doctor_id, date))
            
            for schedule in schedules:
                schedule_dict = {
                    'total_slots': schedule['total_slots'],
                    'booked_slots': schedule['booked_slots'],
                    'time_range': schedule['time_range']
                }
                self.redis.hmset(schedule_key, schedule_dict)
            
            schedule = self.redis.hgetall(schedule_key)
        
        # 计算可用号源
        total_slots = int(schedule[b'total_slots'])
        booked_slots = int(schedule[b'booked_slots'])
        available_slots = total_slots - booked_slots
        
        return {
            'doctor_id': doctor_id,
            'date': date,
            'total_slots': total_slots,
            'booked_slots': booked_slots,
            'available_slots': available_slots,
            'time_range': schedule[b'time_range'].decode()
        }
    
    def book_appointment(self, patient_id, doctor_id, date):
        """预约挂号"""
        # 检查可用号源
        schedule_key = f"doctor:{doctor_id}:schedule:{date}"
        available = self.get_available_slots(doctor_id, date)
        
        if available['available_slots'] <= 0:
            return False, "号源已满"
        
        # 原子操作：挂号
        pipe = self.redis.pipeline()
        pipe.hincrby(schedule_key, 'booked_slots', 1)
        pipe.sadd(f"doctor:{doctor_id}:patients:{date}", patient_id)
        result = pipe.execute()
        
        if result[0] > 0:
            # 生成挂号单
            appointment_id = self.create_appointment(patient_id, doctor_id, date)
            
            # 发送通知
            self.send_notification(patient_id, appointment_id)
            
            return True, {"appointment_id": appointment_id}
        else:
            return False, "挂号失败"
```
